using System;
using System.Collections.Generic;

namespace Allocate_array
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] arrays = new int[20];

            for (int i = 0; i < 20; i++)
            {
                arrays[i] = i * 5;
                Console.Write(arrays[i] + " ");
            }
           
        }
    }
}
